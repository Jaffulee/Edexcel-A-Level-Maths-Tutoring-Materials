\begin{exercise}
Differentiate the following functions.
\begin{enumerate}
    \item $\ln(5x)$
    \item $e^{-20x^2}$
    \item $\sin(16x^3)$
    \item $\cos\!\left(\frac{2}{x}\right)$
\end{enumerate}
\end{exercise}

\begin{exercise}
Evaluate the following indefinite integrals.
\begin{enumerate}
    \item $\displaystyle \int \sin^2 x \, dx$
    \item $\displaystyle \int \cos^2 x \, dx$
\end{enumerate}
\end{exercise}

\begin{exercise}
For each integral below, state a suitable substitution before evaluating the indefinite integral.
\begin{enumerate}
    \item $\displaystyle \int \ln(5x)\,dx$
    \item $\displaystyle \int e^{-20x}\,dx$
    \item $\displaystyle \int \sin(16x)\,dx$
    \item $\displaystyle \int \cos(2x)\,dx$
\end{enumerate}
\end{exercise}
